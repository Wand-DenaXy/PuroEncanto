-- ============================================
-- BASE DE DADOS PUROENCANTO - VERSÃO CORRIGIDA
-- ============================================

CREATE DATABASE IF NOT EXISTS PuroEncanto;
USE PuroEncanto;

-- ============================================
-- CRIAÇÃO DAS TABELAS
-- ============================================

-- Tabela de Tipos de Utilizador
CREATE TABLE TipoUtilizador (
    ID_TipoUtilizador INT PRIMARY KEY AUTO_INCREMENT,
    Tipo VARCHAR(50) NOT NULL
);

-- Tabela de Tipos de Colaboradores
CREATE TABLE TipoColaboradores (
    ID_TipoColaboradores INT PRIMARY KEY AUTO_INCREMENT,
    Tipo VARCHAR(250)
);

-- Tabela de Colaboradores
CREATE TABLE Colaboradores (
    ID_Colaboradores INT PRIMARY KEY AUTO_INCREMENT,
    nome VARCHAR(250),
    telefone VARCHAR(15),  -- CORRIGIDO: VARCHAR em vez de INT
    valor DECIMAL(10,2),   -- CORRIGIDO: DECIMAL em vez de INT
    NIF VARCHAR(250),
    ID_TipoColaboradores INT,
    FOREIGN KEY (ID_TipoColaboradores) REFERENCES TipoColaboradores(ID_TipoColaboradores)
);

-- Tabela de Fornecedores
CREATE TABLE Fornecedores (
    ID_Fornecedor INT PRIMARY KEY AUTO_INCREMENT,
    descricao VARCHAR(100) NOT NULL,
    contacto VARCHAR(15) NOT NULL,
    email VARCHAR(100) NOT NULL,
    morada VARCHAR(100) NOT NULL,
    nif VARCHAR(100) NOT NULL,
    total_debito DECIMAL(15,2)
);

-- Tabela de Total Ativo
CREATE TABLE TotalAtivo (
    ID_TotalAtivo INT PRIMARY KEY AUTO_INCREMENT,
    descricao VARCHAR(100) NOT NULL,
    valor DECIMAL(10,2) NOT NULL,
    data DATE NOT NULL  -- CORRIGIDO: DATE em vez de VARCHAR
);

-- Tabela de Tipos de Eventos
CREATE TABLE TiposEventos (
    ID_TipoEvento INT PRIMARY KEY AUTO_INCREMENT,
    Nome VARCHAR(100) NOT NULL
);

-- Tabela de Pacotes de Convidados
CREATE TABLE PacotesConvidados (
    ID_Pacote INT AUTO_INCREMENT PRIMARY KEY,
    descricao VARCHAR(100) NOT NULL,
    numConvidados INT NOT NULL,
    preco DECIMAL(10,2) NOT NULL
);

-- Tabela de Serviços
CREATE TABLE Servicos (
    ID_Servico INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(50) NOT NULL,
    preco DECIMAL(10,2) NOT NULL
);

-- Tabela de Clientes
CREATE TABLE Clientes (
    ID_Cliente INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    Email VARCHAR(100) NOT NULL,
    nif VARCHAR(100) NOT NULL,
    Password VARCHAR(255) NOT NULL,
    IBAN VARCHAR(255) NOT NULL,
    ID_TipoUtilizador INT DEFAULT 1,
    FOREIGN KEY (ID_TipoUtilizador) REFERENCES TipoUtilizador(ID_TipoUtilizador) ON DELETE SET NULL
);

-- Tabela de Eventos
CREATE TABLE Eventos (
    ID_Evento INT PRIMARY KEY AUTO_INCREMENT,
    ID_Cliente INT NOT NULL,
    Nome VARCHAR(200),
    Data DATE NOT NULL,
    hora TIME NOT NULL,
    estado VARCHAR(50),
    ID_TipoEvento INT,
    ID_Pacote INT,
    precoTotal DOUBLE,
    FOREIGN KEY (ID_TipoEvento) REFERENCES TiposEventos(ID_TipoEvento),
    FOREIGN KEY (ID_Cliente) REFERENCES Clientes(ID_Cliente),
    FOREIGN KEY (ID_Pacote) REFERENCES PacotesConvidados(ID_Pacote)
);

-- Tabela de Fornecimento (CORRIGIDO: removida FK para ID_Produto que não existe)
CREATE TABLE Fornecimento (
    ID_Fornecimento INT PRIMARY KEY AUTO_INCREMENT,
    Data DATE NOT NULL,
    Quantidade INT NOT NULL,
    ID_Fornecedor INT,
    descricao VARCHAR(200),  -- ADICIONADO: descrição do produto
    FOREIGN KEY (ID_Fornecedor) REFERENCES Fornecedores(ID_Fornecedor) ON DELETE CASCADE
);

-- Tabela de Eventos_Servicos
CREATE TABLE Eventos_Servicos (
    ID_Evento INT NOT NULL,
    ID_Servico INT NOT NULL,
    data DATE,
    PRIMARY KEY (ID_Evento, ID_Servico),
    FOREIGN KEY (ID_Evento) REFERENCES Eventos(ID_Evento) ON DELETE CASCADE,
    FOREIGN KEY (ID_Servico) REFERENCES Servicos(ID_Servico) ON DELETE CASCADE
);

-- Tabela de Rendimento
CREATE TABLE Rendimento (
    ID_Rendimento INT PRIMARY KEY AUTO_INCREMENT,
    descricao VARCHAR(100) NOT NULL,
    Valor DECIMAL(10,2) NOT NULL,
    Data DATE NOT NULL
);

-- Tabela de Gastos
CREATE TABLE Gastos (
    ID_Gasto INT PRIMARY KEY AUTO_INCREMENT,
    descricao VARCHAR(100) NOT NULL,
    Valor DECIMAL(10,2) NOT NULL,
    Data DATE NOT NULL
);

-- Tabela de Resumo Financeiro
CREATE TABLE ResumoFinanceiro (
    ID_Finaceiro INT PRIMARY KEY AUTO_INCREMENT,
    descricao VARCHAR(200) NULL,
    ID_Rendimento INT,
    FOREIGN KEY (ID_Rendimento) REFERENCES Rendimento(ID_Rendimento),
    ID_Gasto INT,
    FOREIGN KEY (ID_Gasto) REFERENCES Gastos(ID_Gasto)
);

-- Tabela de Vendas e Serviços
CREATE TABLE VendasServicos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    descricao VARCHAR(100) NOT NULL,
    data DATE NOT NULL,  -- CORRIGIDO: DATE em vez de VARCHAR
    total_vendas DECIMAL(10,2) NOT NULL
);

-- Tabela de Equipamentos
CREATE TABLE Equipamentos (
    ID_Equipamento INT AUTO_INCREMENT PRIMARY KEY,
    descricao VARCHAR(100) NOT NULL,
    valor DECIMAL(10,2) NOT NULL
);

-- Tabela de Finanças
CREATE TABLE Financas (
    ID_Financa INT AUTO_INCREMENT PRIMARY KEY,
    TotalReceber DECIMAL(10,2),
    TotalPagar DECIMAL(10,2),
    Disponibilidades DECIMAL(10,2)
);

-- Tabela de Balancete
CREATE TABLE Balancete (
    ID_Balancete INT AUTO_INCREMENT PRIMARY KEY,
    descricao VARCHAR(100) NOT NULL,
    saldo DECIMAL(15,2)
);

-- Tabela de Economia Financeira
CREATE TABLE econ_financeiro (
    id INT AUTO_INCREMENT PRIMARY KEY,
    tipo VARCHAR(100) NOT NULL,
    sub_tipo VARCHAR(100),
    descricao VARCHAR(255) NOT NULL,
    valor DECIMAL(15,2) NOT NULL,
    ano INT NOT NULL,
    INDEX idx_ano (ano),
    INDEX idx_descricao (descricao)
);

-- Tabela de Demonstração de Resultados
CREATE TABLE demonstracao_resultados (
    id INT AUTO_INCREMENT PRIMARY KEY,
    descricao VARCHAR(255) NOT NULL,
    valor DECIMAL(15,2) NOT NULL,
    ano INT NOT NULL,
    ordem INT NOT NULL,
    tipo VARCHAR(50) DEFAULT 'item',
    INDEX idx_ano (ano),
    INDEX idx_ordem (ordem)
);

-- Tabela de Dívidas a Pagar
CREATE TABLE DividasAPagar (
    ID_Divida INT PRIMARY KEY AUTO_INCREMENT,
    Tipo VARCHAR(50) NOT NULL,
    Valor DECIMAL(15, 2) NOT NULL,
    Estado VARCHAR(50) DEFAULT 'Em aberto',
    ID_Fornecedor INT NULL,
    ID_Cliente INT NULL,
    ID_Colaboradores INT NULL,
    FOREIGN KEY (ID_Fornecedor) REFERENCES Fornecedores(ID_Fornecedor) ON DELETE CASCADE,
    FOREIGN KEY (ID_Cliente) REFERENCES Clientes(ID_Cliente) ON DELETE CASCADE,
    FOREIGN KEY (ID_Colaboradores) REFERENCES Colaboradores(ID_Colaboradores) ON DELETE CASCADE
);

-- 1. Serviços mais utilizados com valor total
SELECT 
    s.nome AS Servico,
    COUNT(es.ID_Servico) AS Quantidade_Vezes_Utilizado,
    SUM(s.preco) AS Valor_Total 
FROM Eventos_Servicos es 
INNER JOIN Servicos s ON es.ID_Servico = s.ID_Servico 
GROUP BY s.ID_Servico, s.nome 
ORDER BY Valor_Total DESC;

-- 2. Serviços mais utilizados por mês
SELECT 
    Servicos.nome AS Servico, 
    COUNT(*) AS Total 
FROM Servicos
INNER JOIN Eventos_Servicos ON Servicos.ID_Servico = Eventos_Servicos.ID_Servico
INNER JOIN Eventos ON Eventos.ID_Evento = Eventos_Servicos.ID_Evento
WHERE MONTH(Eventos.Data) = 4  -- Altere o mês conforme necessário
GROUP BY Servicos.nome
ORDER BY Total DESC;

-- 3. Resumo Financeiro Mensal
SELECT 
    ResumoFinanceiro.descricao AS Meses,
    IFNULL(SUM(Rendimento.valor), 0) - IFNULL(SUM(Gastos.valor), 0) AS Saldo
FROM ResumoFinanceiro
LEFT JOIN Rendimento ON Rendimento.ID_Rendimento = ResumoFinanceiro.ID_Rendimento
LEFT JOIN Gastos ON Gastos.ID_Gasto = ResumoFinanceiro.ID_Gasto
GROUP BY ResumoFinanceiro.descricao, MONTH(Rendimento.Data)
ORDER BY MONTH(Rendimento.Data);

-- 4. Gastos por mês
SELECT
    CASE MONTH(Data)
        WHEN 1 THEN 'Janeiro'
        WHEN 2 THEN 'Fevereiro'
        WHEN 3 THEN 'Março'
        WHEN 4 THEN 'Abril'
        WHEN 5 THEN 'Maio'
        WHEN 6 THEN 'Junho'
        WHEN 7 THEN 'Julho'
        WHEN 8 THEN 'Agosto'
        WHEN 9 THEN 'Setembro'
        WHEN 10 THEN 'Outubro'
        WHEN 11 THEN 'Novembro'
        WHEN 12 THEN 'Dezembro'
    END AS Mes,
    SUM(Valor) AS TotalGastos
FROM Gastos
GROUP BY MONTH(Data)
ORDER BY MONTH(Data);

-- 5. Rendimentos por mês
SELECT
    CASE MONTH(Data)
        WHEN 1 THEN 'Janeiro'
        WHEN 2 THEN 'Fevereiro'
        WHEN 3 THEN 'Março'
        WHEN 4 THEN 'Abril'
        WHEN 5 THEN 'Maio'
        WHEN 6 THEN 'Junho'
        WHEN 7 THEN 'Julho'
        WHEN 8 THEN 'Agosto'
        WHEN 9 THEN 'Setembro'
        WHEN 10 THEN 'Outubro'
        WHEN 11 THEN 'Novembro'
        WHEN 12 THEN 'Dezembro'
    END AS Mes,
    SUM(Valor) AS TotalRendimentos
FROM Rendimento
GROUP BY MONTH(Data)
ORDER BY MONTH(Data);

-- 6. Total de eventos por tipo
SELECT 
    te.Nome AS TipoEvento,
    COUNT(*) AS TotalEventos
FROM Eventos e
INNER JOIN TiposEventos te ON e.ID_TipoEvento = te.ID_TipoEvento
GROUP BY te.Nome
ORDER BY TotalEventos DESC;

-- 7. Clientes com mais eventos
SELECT 
    c.nome AS Cliente,
    COUNT(e.ID_Evento) AS TotalEventos
FROM Clientes c
INNER JOIN Eventos e ON c.ID_Cliente = e.ID_Cliente
GROUP BY c.nome
ORDER BY TotalEventos DESC
LIMIT 10;

-- 8. Vendas por serviço
SELECT 
    descricao, 
    SUM(total_vendas) AS Total
FROM VendasServicos
GROUP BY descricao
ORDER BY Total DESC;

SELECT 
    descricao, 
    total_debito 
FROM Fornecedores
